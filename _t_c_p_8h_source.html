103 public:
104  TCPBase(Inet::TCPEndPoint ** activeConnectionsBuffer, size_t bufferSize, PendingPacket * packetBuffers,
105  size_t packetsBuffersSize) :
106  mActiveConnections(activeConnectionsBuffer),
107  mActiveConnectionsSize(bufferSize), mPendingPackets(packetBuffers), mPendingPacketsSize(packetsBuffersSize)
108  {
109  std::fill(mActiveConnections, mActiveConnections + mActiveConnectionsSize, nullptr);
110  for (size_t i = 0; i < mPendingPacketsSize; ++i)
111  {
112  mPendingPackets[i].peerAddress = PeerAddress::Uninitialized();
113  // In the typical case, the TCPBase constructor is invoked from the TCP constructor on its mPendingPackets,
114  // which has not yet been initialized. That means we can't do a normal move assignment or construction of
115  // the PacketBufferHandle, since that would call PacketBuffer::Free on the uninitialized data.
116  new (&mPendingPackets[i].packetBuffer) System::PacketBufferHandle();
117  }
118  }
