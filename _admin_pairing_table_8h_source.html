117 class ConstAdminIterator

124  ConstAdminIterator(const AdminPairingInfo * start, size_t index, size_t maxSize) :
125  mStart(start), mIndex(index), mMaxSize(maxSize)
126  {
127  if (mIndex >= maxSize)
128  {
129  mIndex = maxSize;
130  }
131  else if (!mStart[mIndex].IsInitialized())
132  {
133  Advance();
134  }
135  }
136  ConstAdminIterator(const ConstAdminIterator &) = default;
137  ConstAdminIterator & operator=(const ConstAdminIterator &) = default;
138 
139  ConstAdminIterator & operator++() { return Advance(); }
140  ConstAdminIterator operator++(int)
141  {
142  ConstAdminIterator other(*this);
143  Advance();
144  return other;
145  }
146 
147  const AdminPairingInfo & operator*() const { return mStart[mIndex]; }
148  const AdminPairingInfo * operator->() const { return mStart + mIndex; }
149 
150  bool operator==(const ConstAdminIterator & other)
151  {
152  if (IsAtEnd())
153  {
154  return other.IsAtEnd();
155  }
156 
157  return (mStart == other.mStart) && (mIndex == other.mIndex) && (mMaxSize == other.mMaxSize);
158  }
159  bool operator!=(const ConstAdminIterator & other) { return !(*this == other); }
160 
161  bool IsAtEnd() const { return (mIndex == mMaxSize); }
162 
163 private:
164  const AdminPairingInfo * mStart;
165  size_t mIndex;
166  size_t mMaxSize;
167 
168  ConstAdminIterator & Advance()
169  {
170  do
171  {
172  if (mIndex < mMaxSize)
173  {
174  mIndex++;
175  }
176  } while (!IsAtEnd() && !mStart[mIndex].IsInitialized());
177 
178  return *this;
179  }
180 };
