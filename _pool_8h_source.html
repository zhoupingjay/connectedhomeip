129  template <typename F>
130  bool ForEachActiveObject(F f)
131  {
132  for (size_t word = 0; word * kBitChunkSize < Capacity(); ++word)
133  {
134  auto & usage = mUsage[word];
135  auto value = usage.load(std::memory_order_relaxed);
136  for (size_t offset = 0; offset < kBitChunkSize && offset + word * kBitChunkSize < Capacity(); ++offset)
137  {
138  if ((value & (kBit1 << offset)) != 0)
139  {
140  if (!f(static_cast<T *>(At(word * kBitChunkSize + offset))))
141  return false;
142  }
143  }
144  }
145  return true;
146  }
147
