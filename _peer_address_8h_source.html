129  void ToString(char * buf, size_t bufSize) const
130  {
131  char ip_addr[kInetMaxAddrLen];
132 
133  switch (mTransportType)
134  {
135  case Type::kUndefined:
136  snprintf(buf, bufSize, "UNDEFINED");
137  break;
138  case Type::kUdp:
139  mIPAddress.ToString(ip_addr);
140  snprintf(buf, bufSize, "UDP:%s:%d", ip_addr, mPort);
141  break;
142  case Type::kTcp:
143  mIPAddress.ToString(ip_addr);
144  snprintf(buf, bufSize, "TCP:%s:%d", ip_addr, mPort);
145  break;
146  case Type::kBle:
147  // Note that BLE does not currently use any specific address.
148  snprintf(buf, bufSize, "BLE");
149  break;
150  default:
151  snprintf(buf, bufSize, "ERROR");
152  break;
153  }
154  }
