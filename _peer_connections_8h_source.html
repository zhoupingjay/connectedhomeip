337  template <typename Callback>
338  void ExpireInactiveConnections(uint64_t maxIdleTimeMs, Callback callback)
339  {
340  const uint64_t currentTime = mTimeSource.GetCurrentMonotonicTimeMs();
341 
342  for (size_t i = 0; i < kMaxConnectionCount; i++)
343  {
344  if (!mStates[i].GetPeerAddress().IsInitialized())
345  {
346  continue; // not an active connection
347  }
348 
349  uint64_t connectionActiveTime = mStates[i].GetLastActivityTimeMs();
350  if (connectionActiveTime + maxIdleTimeMs >= currentTime)
351  {
352  continue; // not expired
353  }
354 
355  MarkConnectionExpired(&mStates[i], callback);
356  }
357  }
